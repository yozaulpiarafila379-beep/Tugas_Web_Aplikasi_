📝 Pseudocode — Algoritma Sistem

Berikut adalah pseudocode untuk setiap class. Klik tombol ▶ Tampilkan DartPad untuk melihat kode implementasi di Dart.

1. Abstract Class EntitasGudang

// ABSTRACT CLASS: EntitasGudang
abstract class EntitasGudang {
  String kode
  String nama
  constructor EntitasGudang(kode, nama)
  abstract void tampilkanInfo()
  void cetakIdentitas()
}

2. Class Barang (extends EntitasGudang)

// CLASS: Barang
class Barang extends EntitasGudang {
  String kategori
  String merk
  int jumlahStok
  String satuan
  String kondisi
  constructor Barang(kode, nama, kategori, merk, stok, satuan, kondisi)
  @override void tampilkanInfo()
  void tambahStok(int jumlah)
  void kurangiStok(int jumlah)
}

3. Abstract Class Transaksi

// ABSTRACT CLASS: Transaksi
abstract class Transaksi {
  String idTransaksi
  String tanggal
  String kodeBarang
  int jumlah
  constructor Transaksi(id, tgl, kode, jml)
  abstract void prosesTransaksi()
  void cetakHeader()
}

4. Class BarangMasuk (extends Transaksi)

// CLASS: BarangMasuk
class BarangMasuk extends Transaksi {
  String namaBarang
  constructor BarangMasuk(id, tgl, kode, jml, nama)
  @override void prosesTransaksi() {
    barang = cariBarang(kodeBarang)
    if (barang != null) { barang.tambahStok(jumlah); catatRiwayat() }
    else print("❌ Barang tidak ditemukan!")
  }
}

5. Class BarangKeluar (extends Transaksi)

// CLASS: BarangKeluar
class BarangKeluar extends Transaksi {
  String tujuan
  constructor BarangKeluar(id, tgl, kode, jml, tujuan)
  @override void prosesTransaksi() {
    barang = cariBarang(kodeBarang)
    if (barang != null) { barang.kurangiStok(jumlah); catatRiwayat() }
    else print("❌ Barang tidak ditemukan!")
  }
}

6. Class StokBarang (Perhitungan Stok)

// CLASS: StokBarang
class StokBarang {
  int stokAwal
  int barangMasuk
  int barangKeluar
  constructor StokBarang(awal, masuk, keluar)
  void hitungStok() {
    stokAkhir = stokAwal + barangMasuk - barangKeluar
    if (stokAkhir == 0) status = "Habis"
    else if (stokAkhir < 5) status = "Menipis"
    else status = "Aman"
    print("📊 Stok Akhir: " + stokAkhir + " (" + status + ")")
  }
}

🧬 4 Pilar OOP yang Diterapkan

1. Enkapsulasi

Status internal barang (stok, kondisi) hanya diubah lewat metode tambahStok() dan kurangiStok().

2. Pewarisan (Inheritance)

Barang extends EntitasGudang. BarangMasuk dan BarangKeluar extends Transaksi.

3. Polimorfisme

tampilkanInfo() dan prosesTransaksi() di-override pada class turunan.

4. Interaksi Objek

Objek Barang berinteraksi dengan StokBarang dan Transaksi untuk mengelola stok.

abstract class EntitasGudang {
String kode;
String nama;

EntitasGudang(this.kode, this.nama);

void tampilkanInfo();

void cetakIdentitas() {
print("🔹 $kode — $nama");
}
}

class Barang extends EntitasGudang {
String kategori;
String merk;
int jumlahStok;
String satuan;
String kondisi;

Barang(
String kode,
String nama,
this.kategori,
this.merk,
this.jumlahStok,
this.satuan,
this.kondisi,
) : super(kode, nama);

@override
void tampilkanInfo() {
print("📦 $kode | $nama | $kategori | Stok: $jumlahStok $satuan");
}

void tambahStok(int jumlah) {
jumlahStok += jumlah;
print("✅ Stok bertambah $jumlah. Total: $jumlahStok");
}

void kurangiStok(int jumlah) {
if (jumlahStok >= jumlah) {
jumlahStok -= jumlah;
print("✅ Stok berkurang $jumlah. Total: $jumlahStok");
} else {
print("❌ Stok tidak mencukupi!");
}
}
}
